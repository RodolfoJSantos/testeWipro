SELECT COUNT(1) AS quantidade, st.dsStatus 
FROM tb_Processo pro
right join tb_status st
ON pro.idStatus = st.idStatus
GROUP BY st.dsStatus

SELECT MAX(dtAndamento) MaiorDtAndamento FROM tb_Andamento andm
inner join tb_Processo pro
ON andm.idProcesso = pro.IdProcesso
WHERE YEAR(pro.[DtEncerramento]) = 2013

SELECT COUNT(DtEncerramento) Quantidade FROM tb_Processo
GROUP BY DtEncerramento
HAVING COUNT(DtEncerramento) > 5
